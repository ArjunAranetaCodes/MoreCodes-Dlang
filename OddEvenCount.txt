/*
Write a program to count the number of even and odd numbers 
from a series of numbers from five(5) number inputs of the user.
*/

import std.stdio;

void main(){
	int numInput = 0, evenCount = 0, oddCount = 0;
	
	for(int i=0; i<5; i++){
		write("Enter a number: ");
		scanf("%d", &numInput);	
		
		if(numInput % 2 == 0){
			evenCount = evenCount + 1;
		}else{
			oddCount = oddCount + 1;
		}
	}
	
	writeln("There are ", evenCount, " even numbers.");
	writeln("There are ", oddCount, " odd numbers.");
}
